# 问题：箭头函数的this是如何查找的？

# 答案：向外层作用域中，一层层查找this，知道有this的定义。

// vue
const obj = {
    aaa() {
        setTimeout(function () {
            setTimeout(function () {
                console.log(this); // window
            })
            
            setTimeout(() => {
                cosole.log(this); // window
            })
        })
        
        setTimeout(() => {
            setTimeout(function () {
                console.log(this); // window
            })
            
            setTimeout(() => {
                console.log(this); // aaa()
            })
        })
    }
};
obj.aaa();

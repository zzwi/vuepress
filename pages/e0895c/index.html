# call_apply_bind的简单实现

# 手写简单的call_apply_bind(不考虑edge case)

# call

// JavaScript
// call
// 给函数原型添加zycall()
Function.prototype.zycall = function (thisArgs, ...args) {
    // 1. 给thisArgs作判断和对象类型转换(防止它传入的是非对象类型)
    // 使数字0能被转换为Number对象
    thisArgs = (thisArgs !== undefined && thisArgs !== null) ? Object(thisArgs) : window;
    // 2. 绑定上层作用域的this
    thisArgs.fn = this;
    // 3. 调用需要被执行的函数并赋值给result
    var result = thisArgs.fn(...args);
    delete thisArgs.fn;
    // 4. 将执行结果返回出去
    return result;
};

function foo() {
    console.log("函数foo成功调用",this);
};
function sum(num1,num2) {
    console.log("函数sum成功调用",this,num1,num2);
};

foo.zycall(123); // 函数foo成功调用 Number {123, fn: ƒ}
sum.zycall('123',10,20); // 函数sum成功调用 String {'123', fn: ƒ} 10 20

# apply

// JavaScript
// apply
Function.prototype.zyapply = function (thisArgs,argArray) {
    // 1. 给thisArgs作判断和对象类型转换(防止它传入的是非对象类型)
    thisArgs = (thisArgs !== undefined && thisArgs !== null) ? Object(thisArgs) : window;
    // 2. 绑定上层作用域的this
    thisArgs.fn = this;
    // 3. 判断argArray是否为空
    argArray = argArray || [];
    // 4. 调用需要被执行的函数并赋值给result
    var result = thisArgs.fn(...argArray);
    delete thisArgs.fn;
    // 5. 将执行结果返回出去
    return result;
};

function foo() {
    console.log("函数foo成功调用",this);
};
function sum(num1,num2) {
    console.log("函数sum成功调用",this,num1,num2);
};

foo.zyapply(0); // 函数foo成功调用 Number {0, fn: ƒ}
sum.zyapply('123',[30,40]); // 函数sum成功调用 String {'123', fn: ƒ} 30 40

# bind

// JavaScript
// bind
Function.prototype.zybind = function (thisArg, ...argArray) {
    // 1. 获取到真实需要调用的函数
    var fn = this;
    // 2. 构造新函数
    function proxyFn(...args){
        // 3. 给thisArgs作判断和对象类型转换(防止它传入的是非对象类型)
        thisArgs = (thisArgs !== undefined && thisArgs !== null) ? Object(thisArgs) : window;
        // 4. 整合参数
        var finalArgs = [...argArray,...args];
        // 5. 调用需要被执行的函数并赋值给result
        // ...finalArgs 将数组中的参数逐个提取出来
        var result = thisArgs.fn(...finalArgs);
        delete thisArgs.fn;
        // 6. 将执行结果返回出去
        return result;
    };
    // 7. 将新函数返回出去
    return proxyFn;
};

function foo() {
    console.log("函数foo成功调用",this);
};
function sum(...num) {
    console.log("函数sum成功调用",this,...num);
};

var bar = foo.bind('abc');
bar(); // 函数foo成功调用 String {'abc'}
var baz = sum.bind(123,50,60);
baz(70,80); // 函数sum成功调用 Number {'123'} 50 60 70 80

# arguments的基本使用

直接上代码吧

// JavaScript
function foo(num1,num2){
    // 类数组对象(看上去是个数组，实际是对象)：arguments
    console.log(arguments); // Arguments(5)
    // 常见的对arguments的三个操作
    // 1. 获取arguments的长度
    console.log(arguments.length); // 5
    // 2. 通过下标获取arguments的元素
    console.log(arguments[0]); // 10
    console.log(arguments[1]); // 20
    // 3. 获取调用arguments的原函数
    console.log(arguments.callee);
};

foo(10, 20, 30, 40, 50);

# arguments转array

直接放代码

// JavaScript
function foo(num1,num2){
    // arguments转成array的三种方法
    // 1. 遍历自己
    var newArr1 = [];
    for(var i = 0; i < arguments.length; i++){
        newArr1.push(arguments[i]);
    };
    console.log(newArr1);// [10, 20, 30, 40, 50]

    // 2. Array.prototype.slice将arguments转成array
    var newArr2_1 = Array.prototype.slice.call(arguments);
    console.log(newArr2_1);// [10, 20, 30, 40, 50]
    var newArr2_2 = [].slice.call(arguments);
    console.log(newArr2_2);// [10, 20, 30, 40, 50]

    // 3. ES6的写法
    var newArr3_1 = Array.from(arguments);
    console.log(newArr3_1);// [10, 20, 30, 40, 50]
    var newArr3_2 = [...arguments];
    console.log(newArr3_2);// [10, 20, 30, 40, 50]
};
foo(10, 20, 30, 40, 50);

# 箭头函数没有arguments

直接看下面代码

# 手写Array简单的slice方法(不考虑edge case)

// JavaScript
Array.prototype.zyslice = function (start, end) {
    start = start || 0;
    end = end || this.length;
    // 1. 获取需要执行的函数
    var arr = this;
    var newArr = [];
    // 2. 遍历
    for(var i = start; i < end; i++){
        newArr.push(arr[i])
    };
    // 3. 将新数组返回
    return newArr;
};

function foo(num1,num2){
    console.log(arguments);// Arguments [1, 2, 3, 4, 5]
    console.log(Array.prototype.zyslice.call(arguments));
    // Array [1, 2, 3, 4, 5]
};

foo(1, 2, 3, 4, 5);

var bar = Array.prototype.zyslice.call([6, 7, 8, 9], 1, 3);
console.log(bar);// Array [7, 8]

var baz = ['a','b','c','d','e'];
console.log(baz.zyslice(1,5));// Array [b, c, d, e]

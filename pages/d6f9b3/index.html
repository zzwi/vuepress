// JavaScript(node环境)
class ZyPromise {
  constructor(executor) {
    executor()

# 传入resolve和reject

既是executor的两个参数，又是两个单独的函数，也就是回调函数。

有着其中一个先被调用那么另一个就失效的特点

# 1. resolve和reject的调用

回调函数的简单调用，不包含参数和互斥，也就是

  1. 没有参数
  2. 同时调用
// JavaScript(node环境)
class ZyPromise {
  constructor(executor) {
    const resolve = () => {
      console.log("resolve被调用")
    }
    const reject = () => {
      console.log("reject被调用")
    }
    executor(resolve, reject)
  }
}

const promise = new ZyPromise((resolve, reject) => {
  // console.log("自制的ZyPromise被调用")
  resolve()
  reject()
})
/**
  resolve被调用
  reject被调用
 */

# 2.resolve和reject的参数

让resolve和reject这两个回调函数的参数得以传入自制的ZyPromise，

并保存下来，为后面then等其他方法的参数(res,err)传入做准备。

// JavaScript(node环境)
class ZyPromise {
  constructor(executor) {
    this.value = undefined
    this.reason = undefined
    const resolve = (value) => {
      this.value = value
      console.log("resolve被调用", value)
    }
    const reject = (reason) => {
      this.reason = reason
      console.log("reject被调用", reason)
    }
    executor(resolve, reject)
  }
}

const promise = new ZyPromise((resolve, reject) => {
  // console.log("自制的ZyPromise被调用")
  resolve(111)
  reject(222)
})
/**
  resolve被调用 111
  reject被调用 222

# 3.resolve和reject的互斥

这两个函数，当其中一个先被调用时，另一个就得失效。

原因在于Promise的状态：

当执行executor而未执行resolve或reject时，状态为pending，此时不影响resolve或reject的执行

当先执行resolve时，状态为fulfilled，此时状态固定且不能再执行reject，但不影响其他普通语句的执行

当先执行reject时，状态为rejected，此时状态固定且不能再执行resolve，但不影响其他普通语句的执行

// JavaScript(node环境)
// 状态常量
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_FULFILLED
        this.value = value
        console.log("resolve被调用", value)
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_REJECTED
        this.reason = reason
        console.log("reject被调用", reason)
      }
    }
    executor(resolve, reject)
  }
}

const promise = new ZyPromise((resolve, reject) => {
  // console.log("自制的ZyPromise被调用")
  resolve(111)
  reject(222)
})
/**
  resolve被调用 111
 */

# 实现-简单then方法

在自制的ZyPromise类中写入实例方法then，接收resolve和reject传入的参数。

# 1.then方法的简单写入

// JavaScript(node环境)
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_FULFILLED
        this.value = value
        console.log("resolve被调用", value)
        this.onfulfilled(this.value)
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_REJECTED
        this.reason = reason
        console.log("reject被调用", reason)
        this.onrejected(this.reason)
      }
    }
    executor(resolve, reject)
  }

  then(onfulfilled, onrejected) {
    this.onfulfilled = onfulfilled
    this.onrejected = onrejected
  }
}

const promise = new ZyPromise((resolve, reject) => {
  // console.log("自制的ZyPromise被调用")
  resolve(111)
  reject(222)
})
// 调用then方法
promise.then(res => {
  console.log("res: ", res)
}, err => {
  console.log("err: ", err)
})

如上代码所示，它报错了。

原因是代码执行顺序，当执行到resolve方法内的 this.onfulfilled 时，下面的then方法还未执行，所以此时 this.onfulfilled 的值依然是最初的 null 。

解决办法：

延迟 resolve 和 reject 的执行

可以分别从宏任务和微任务出发进行延迟

宏任务可以使用 setTimeout 方法，

微任务可以使用 queueMicrotask 方法，

都可以完成目的，但是从微任务角度入手更好，

即使用 queueMicrotask 方法效果更好

// JavaScript(node环境)
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_FULFILLED
        // 添加微任务
        queueMicrotask(() => {
          // this.status = PROMISE_STATUS_FULFILLED
          this.value = value
          console.log("resolve被调用", value)
          this.onfulfilled(this.value)
        })
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_REJECTED
        // 添加微任务
        queueMicrotask(() => {
          // this.status = PROMISE_STATUS_REJECTED
          this.reason = reason
          console.log("reject被调用", reason)
          this.onrejected(this.reason)
        })
      }
    }
    executor(resolve, reject)
  }

  then(onfulfilled, onrejected) {
    this.onfulfilled = onfulfilled
    this.onrejected = onrejected
  }
}

const promise = new ZyPromise((resolve, reject) => {
  // console.log("自制的ZyPromise被调用")
  resolve(111)
  reject(222)
})
// 调用then方法
promise.then(res => {
  console.log("res: ", res)
}, err => {
  console.log("err: ", err)
})
/**
  resolve被调用 111
  res:  111

# 2.then方法的简单优化

实现then方法的多次调用、确定Promise状态之后的延时调用

以及保证executor回调函数中参数数量只有1个时也能传入并调用

# 实现then方法的多次调用

// JavaScript(node环境)
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined
    this.onfulfilledFns = []
    this.onrejectedFns = []

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_FULFILLED
        this.value = value
        // 添加微任务
        queueMicrotask(() => {
          // this.status = PROMISE_STATUS_FULFILLED
          // this.value = value
          this.onfulfilledFns.forEach(fn => {
            fn(this.value)
          })
        })
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        this.status = PROMISE_STATUS_REJECTED
        this.reason = reason
        // 添加微任务
        queueMicrotask(() => {
          // this.status = PROMISE_STATUS_REJECTED
          // this.reason = reason
          this.onrejectedFns.forEach(fn => {
            fn(this.reason)
          })
        })
      }
    }
    executor(resolve, reject)
  }

  then(onfulfilled, onrejected) {
    // 1.如果在then调用的时候，状态已经确定下来
    if(this.status === PROMISE_STATUS_FULFILLED && onfulfilled) {
      onfulfilled(this.value)
    }
    if(this.status === PROMISE_STATUS_REJECTED && onrejected) {
      onrejected(this.reason)
    }

    // 2.将resolve回调和reject回调分别放到数组中
    if(this.status === PROMISE_STATUS_PENDING && onfulfilled) {
      this.onfulfilledFns.push(onfulfilled)
    }
    if(this.status === PROMISE_STATUS_PENDING && onrejected) {
      this.onrejectedFns.push(onrejected)
    }
  }
}

const promise = new ZyPromise((resolve, reject) => {
  resolve(111)
  reject(222)
})

// 调用then方法-多次调用
// 第一次
promise.then(res => {
  console.log("res1: ", res)
}, err => {
  console.log("err1: ", err)
})
// 第二次
promise.then(res => {
  console.log("res2: ", res)
}, err => {
  console.log("err2: ", err)
})

// 在确定Promise状态之后，再次调用then
setTimeout(() => {
  promise.then(res => {
    console.log("res3: ", res)
  })
},1000)
/**
  res1:  111
  res2:  111
  res3:  111
 */

# 实现then方法的链式调用

// JavaScript(node环境)
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

// 工具函数
function execFunctionWithCatchError(execFn, value, resolve, reject) {
  try {
    const result = execFn(value)
    resolve(result)
  } catch(err) {
    reject(err)
  }
}


class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined
    this.onfulfilledFns = []
    this.onrejectedFns = []

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        // 添加微任务
        queueMicrotask(() => {
          if(this.status !== PROMISE_STATUS_PENDING) return
          this.status = PROMISE_STATUS_FULFILLED
          this.value = value
          this.onfulfilledFns.forEach(fn => {
            fn(this.value)
          })
        })
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        // 添加微任务
        queueMicrotask(() => {
          if(this.status !== PROMISE_STATUS_PENDING) return
          this.status = PROMISE_STATUS_REJECTED
          this.reason = reason
          this.onrejectedFns.forEach(fn => {
            fn(this.reason)
          })
        })
      }
    }

    try{
      executor(resolve, reject)
    }catch(err) {
      reject(err)
    }
  }

  then(onfulfilled, onrejected) {
    return new ZyPromise((resolve, reject) => {
      // 1.如果在then调用的时候，状态已经确定下来
      if(this.status === PROMISE_STATUS_FULFILLED && onfulfilled) {
        // try{
        //   const value = onfulfilled(this.value)
        //   resolve(value)
        // }catch(err) {
        //   reject(err)
        // }
        execFunctionWithCatchError(onfulfilled, this.value, resolve, reject)
      }
      if(this.status === PROMISE_STATUS_REJECTED && onrejected) {
        // try{
        //   const reason = onrejected(this.reason)
        //   resolve(reason)
        // }catch(err) {
        //   reject(err)
        // }
        execFunctionWithCatchError(onrejected, this.reason, resolve, reject)
      }

      // 2.将resolve回调和reject回调分别放到数组中
      if(this.status === PROMISE_STATUS_PENDING && onfulfilled) {
        this.onfulfilledFns.push(() => {
          // try{
          //   const value = onfulfilled(this.value)
          //   resolve(value)
          // }catch(err) {
          //   reject(err)
          // }
          execFunctionWithCatchError(onfulfilled, this.value, resolve, reject)
        })
      }
      if(this.status === PROMISE_STATUS_PENDING && onrejected) {
        this.onrejectedFns.push(() => {
          // try{
          //   const reason = onrejected(this.reason)
          //   resolve(reason)
          // }catch(err) {
          //   reject(err)
          // }
          execFunctionWithCatchError(onrejected, this.reason, resolve, reject)
        })
      }
    })
  }
}

const promise = new ZyPromise((resolve, reject) => {
  resolve(111)
  reject(222)
})

// then方法-链式调用
promise.then(res => {
  console.log("res1: ",res)
  return "aaa"
  // throw new Error("res1 Error")
}).then(res => {
  console.log("res2: ",res)
  return "bbb"
}, err => {
  console.log("err: ", err)
})

  res1:  111
  res2:  aaa

# 实现-简单catch方法

执行reject回调，接收reject传递的参数

// JavaScript(node环境)
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

// 工具函数
function execFunctionWithCatchError(execFn, value, resolve, reject) {
  try {
    const result = execFn(value)
    resolve(result)
  } catch(err) {
    reject(err)
  }
}


class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined
    this.onfulfilledFns = []
    this.onrejectedFns = []

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        // 添加微任务
        queueMicrotask(() => {
          if(this.status !== PROMISE_STATUS_PENDING) return
          this.status = PROMISE_STATUS_FULFILLED
          this.value = value
          this.onfulfilledFns.forEach(fn => {
            fn(this.value)
          })
        })
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        // 添加微任务
        queueMicrotask(() => {
          if(this.status !== PROMISE_STATUS_PENDING) return
          this.status = PROMISE_STATUS_REJECTED
          this.reason = reason
          this.onrejectedFns.forEach(fn => {
            fn(this.reason)
          })
        })
      }
    }

    try{
      executor(resolve, reject)
    }catch(err) {
      reject(err)
    }
  }

  // then方法设计
  then(onfulfilled, onrejected) {
    onrejected = onrejected || (err => { throw err })

    return new ZyPromise((resolve, reject) => {
      // 1.如果在then调用的时候，状态已经确定下来
      if(this.status === PROMISE_STATUS_FULFILLED && onfulfilled) {
        execFunctionWithCatchError(onfulfilled, this.value, resolve, reject)
      }
      if(this.status === PROMISE_STATUS_REJECTED && onrejected) {
        execFunctionWithCatchError(onrejected, this.reason, resolve, reject)
      }

      // 2.将resolve回调和reject回调分别放到数组中
      if(this.status === PROMISE_STATUS_PENDING && onfulfilled) {
        this.onfulfilledFns.push(() => {
          execFunctionWithCatchError(onfulfilled, this.value, resolve, reject)
        })
      }
      if(this.status === PROMISE_STATUS_PENDING && onrejected) {
        this.onrejectedFns.push(() => {
          execFunctionWithCatchError(onrejected, this.reason, resolve, reject)
        })
      }
    })
  }

  // catch方法设计
  catch(onrejected) {
    this.then(undefined, onrejected)
  }
}

const promise = new ZyPromise((resolve, reject) => {
  // resolve(111)
  reject(222)
})

// then、catch方法
promise.then(res => {
  console.log("res1: ",res)
  return "aaa"
}).then(res => {
  console.log("res2: ",res)
}).catch(err => {
  console.log("err1: ",err)
})
// err1:  222

# 实现-简单finally方法

无论Promise是处于fulfilled状态还是rejected状态，最后都会执行finally方法。

// JavaScript(node环境)
const PROMISE_STATUS_PENDING = 'pending'
const PROMISE_STATUS_FULFILLED = 'fulfilled'
const PROMISE_STATUS_REJECTED = 'rejected'

// 工具函数
function execFunctionWithCatchError(execFn, value, resolve, reject) {
  try {
    const result = execFn(value)// 注意：这里是会自动执行一次execFn函数的！
    resolve(result)
  } catch(err) {
    reject(err)
  }
}


class ZyPromise {
  constructor(executor) {
    this.status = PROMISE_STATUS_PENDING
    this.value = undefined
    this.reason = undefined
    this.onfulfilledFns = []
    this.onrejectedFns = []

    const resolve = (value) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        // 添加微任务
        queueMicrotask(() => {
          if(this.status !== PROMISE_STATUS_PENDING) return
          this.status = PROMISE_STATUS_FULFILLED
          this.value = value
          this.onfulfilledFns.forEach(fn => {
            fn(this.value)
          })
        })
      }
    }
    const reject = (reason) => {
      if(this.status === PROMISE_STATUS_PENDING) {
        // 添加微任务
        queueMicrotask(() => {
          if(this.status !== PROMISE_STATUS_PENDING) return
          this.status = PROMISE_STATUS_REJECTED
          this.reason = reason
          this.onrejectedFns.forEach(fn => {
            fn(this.reason)
          })
        })
      }
    }

    try{
      executor(resolve, reject)
    }catch(err) {
      reject(err)
    }
  }

  // then方法设计
  then(onfulfilled, onrejected) {
    onrejected = onrejected || (err => { throw err })
    onfulfilled = onfulfilled || (value => { return value })

    return new ZyPromise((resolve, reject) => {
      // 1.如果在then调用的时候，状态已经确定下来
      if(this.status === PROMISE_STATUS_FULFILLED && onfulfilled) {
        execFunctionWithCatchError(onfulfilled, this.value, resolve, reject)
      }
      if(this.status === PROMISE_STATUS_REJECTED && onrejected) {
        execFunctionWithCatchError(onrejected, this.reason, resolve, reject)
      }

      // 2.将resolve回调和reject回调分别放到数组中
      if(this.status === PROMISE_STATUS_PENDING && onfulfilled) {
        this.onfulfilledFns.push(() => {
          execFunctionWithCatchError(onfulfilled, this.value, resolve, reject)
        })
      }
      if(this.status === PROMISE_STATUS_PENDING && onrejected) {
        this.onrejectedFns.push(() => {
          execFunctionWithCatchError(onrejected, this.reason, resolve, reject)
        })
      }
    })
  }

  // catch方法设计
  catch(onrejected) {
    return this.then(undefined, onrejected)
  }

  // finally方法设计
  finally(onFinally) {
    this.then(() => {
      onFinally()
    }, () => {
      onFinally()
    })
  }
}

# 实现-all_allSettled方法

在自制ZyPromise中加入类方法all和allSettled

all 方法可以将收纳多个ZyPromise的数组作为参数，当这些ZyPromise都为fulfilled状态时，all方法会等所有ZyPromise都得到结果时再执行

当这些ZyPromise中出现第一个rejected状态时，all方法会直接返回第一个reject的值，并不再执行。

allSettled 方法同样可以将收纳多个ZyPromise的数组作为参数，并且无论这些ZyPromise状态如何，allSettled方法都会等所有ZyPromise都得到结果时再执行，allSettled的返回状态永远是fulfilled，它会返回一个数组，数组内有一个个包含每个ZyPromise的状态(status)和返回值(res/err)的对象。
